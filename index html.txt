    // --- نظام النطق الصوتي ---
    function speakName(name, cls) {
        if ('speechSynthesis' in window) {
            const text = `تنبيه، خروج الطالب ${name} من الصف ${cls}`;
            const msg = new SpeechSynthesisUtterance(text);
            msg.lang = 'ar-SA';
            msg.rate = 0.8;
            window.speechSynthesis.speak(msg);
        }
    }

    // --- وظائف الإدارة ---
    function uploadData(type, event) {
        const file = event.target.files[0];
        const reader = new FileReader();
        reader.onload = (e) => {
            const data = new Uint8Array(e.target.result);
            const workbook = XLSX.read(data, {type: 'array'});
            const jsonData = XLSX.utils.sheet_to_json(workbook.Sheets[workbook.SheetNames[0]]);
            db.ref('school_config/' + type).set(jsonData)
                .then(() => alert("تم تحديث البيانات بنجاح!"));
        };
        reader.readAsArrayBuffer(file);
    }

    // --- نظام ولي الأمر ---
    function parentLogin() {
        const qid = document.getElementById('parentQID').value;
        if(!qid) return alert("يرجى إدخال الرقم الشخصي");

        db.ref('school_config/students').once('value', (snap) => {
            const students = snap.val() || [];
            const myKids = students.filter(s => s.QID_ولي_الأمر == qid);

            if(myKids.length > 0) {
                document.getElementById('parent-login').style.display = 'none';
                document.getElementById('parent-portal').style.display = 'block';
                document.getElementById('parent-welcome').innerText = "مرحباً: " + myKids[0].اسم_ولي_الأمر;
                
                const container = document.getElementById('children-container');
                container.innerHTML = "";
                myKids.forEach(kid => {
                    container.innerHTML += `
                        <div class="child-card" onclick="sendCall('${kid.اسم_الطالب}', '${kid.صف}')">
                            <div><strong>${kid.اسم_الطالب}</strong><br><small>الصف: ${kid.صف}</small></div>
                            <span style="color:var(--primary)">ارسال نداء ←</span>
                        </div>`;
                });
            } else { alert("الرقم الشخصي غير مسجل."); }
        });
    }

    function sendCall(name, cls) {
        db.ref('live_calls/' + cls).push({
            name: name,
            cls: cls,
            time: new Date().toLocaleTimeString('ar-EG'),
            date: new Date().toLocaleDateString('ar-EG'),
            ts: Date.now()
        });
        alert("تم إرسال النداء للمعلم.");
    }

    // --- نظام المعلم ---
    let myClasses = [];
    function teacherLogin() {
        const user = document.getElementById('teacherUsername').value;
        db.ref('school_config/teachers').once('value', (snap) => {
            const teachers = snap.val() || [];
            const teacher = teachers.find(t => t.Username == user);

            if(teacher) {
                document.getElementById('teacher-login').style.display = 'none';
                document.getElementById('teacher-portal').style.display = 'block';
                document.getElementById('teacher-welcome-msg').innerText = "أهلاً أ/ " + teacher.اسم_المعلم;
                myClasses = teacher.الصفوف.split(',').map(s => s.trim());
                document.getElementById('teacher-classes-list').innerText = "صفوفك المتابعة: " + myClasses.join(' | ');

                myClasses.forEach(cls => {
                    db.ref('live_calls/' + cls).on('child_added', (newSnap) => {
                        const call = newSnap.val();
                        document.getElementById('notifSound').play();
                        speakName(call.name, cls);
                        refreshTeacherView();
                    });
                    db.ref('live_calls/' + cls).on('child_removed', refreshTeacherView);
                });
            } else { alert("المستخدم غير موجود."); }
        });
    }

    function refreshTeacherView() {
        const container = document.getElementById('teacher-requests-container');
        container.innerHTML = "";
        db.ref('live_calls').once('value', (snap) => {
            const all = snap.val() || {};
            myClasses.forEach(cls => {
                if(all[cls]) {
                    for(let id in all[cls]) {
                        const c = all[cls][id];
                        container.innerHTML += `
                            <div class="request-item">
                                <div><strong>${c.name}</strong> (${cls})</div>
                                <button class="btn btn-success" style="width:auto" onclick="finishCall('${cls}', '${id}')">تم الخروج</button>
                            </div>`;
                    }
                }
            });
        });
    }

    function finishCall(cls, id) {
        db.ref('live_calls/' + cls + '/' + id).once('value', (snap) => {
            db.ref('reports').push(snap.val());
            db.ref('live_calls/' + cls + '/' + id).remove();
        });
    }

    // --- التقارير ---
    db.ref('reports').on('value', (snap) => {
        const reports = snap.val();
        const body = document.getElementById('report-body');
        body.innerHTML = "";
        for(let id in reports) {
            const r = reports[id];
            body.innerHTML += `<tr><td>${r.name}</td><td>${r.cls}</td><td>${r.time}</td><td>${r.date}</td></tr>`;
        }
    });

    function downloadReport() {
        db.ref('reports').once('value', (snap) => {
            const ws = XLSX.utils.json_to_sheet(Object.values(snap.val() || {}));
            const wb = XLSX.utils.book_new();
            XLSX.utils.book_append_sheet(wb, ws, "المناداة");
            XLSX.writeFile(wb, "تقرير_المدرسة.xlsx");
        });
    }

    // --- عام ---
    function showScreen(id) {
        document.querySelectorAll('.view-content').forEach(v => v.classList.remove('active'));
        document.querySelectorAll('.nav-tab').forEach(n => n.classList.remove('active'));
        document.getElementById(id + '-screen').classList.add('active');
        document.getElementById('tab-' + id).classList.add('active');
    }

    setInterval(() => { document.getElementById('clock').innerText = new Date().toLocaleTimeString('ar-EG'); }, 1000);
